from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score
from sklearn.model_selection import KFold

def train(k, X_train, y_train):
    knn = KNeighborsClassifier(n_neighbors=k)
    knn.fit(X_train, y_train)
    return knn

def test(model, X_test, y_test):
    predictions = model.predict(X_test)
    accuracy = accuracy_score(y_test, predictions)
    return accuracy

def cross_validate(k):
    cv_acc_scores = []
    kf = KFold(n_splits=4, shuffle=True, random_state=1)  # Using 4 folds.
    for train_index, test_index in kf.split(features):
        X_train, X_test = features.loc[train_index], features.loc[test_index]
        y_train, y_test = labels.loc[train_index], labels.loc[test_index]
        model = train(k, X_train, y_train)
        accuracy = test(model, X_test, y_test)
        cv_acc_scores.append(accuracy)
        avg_score = np.mean(cv_acc_scores)
    return np.round(avg_score, 3)

from sklearn.neural_network import MLPClassifier
from sklearn.utils._testing import ignore_warnings
from sklearn.exceptions import ConvergenceWarning

def train(hidden_layer_sizes, X_train, y_train):
    mlp = MLPClassifier(hidden_layer_sizes)
    mlp.fit(X_train, y_train)
    return mlp

def test(model, X_test, y_test):
    predictions = model.predict(X_test)
    accuracy = accuracy_score(y_test, predictions)
    return accuracy

@ignore_warnings(category=ConvergenceWarning)
def cross_validate(hidden_layer_sizes):
    
    cv_acc_scores_trn = []  # Now we collect train and test accuracies.
    cv_acc_scores_tst = []  # Comparing them checks for overfitting. 
    
    kf = KFold(n_splits=4, shuffle=True, random_state=1)  # Using 4 folds.
    
    for train_index, test_index in kf.split(features):
        X_train, X_test = features.loc[train_index], features.loc[test_index]
        y_train, y_test = labels.loc[train_index], labels.loc[test_index]
        model = train(hidden_layer_sizes, X_train, y_train)
        
        accuracy_trn = test(model, X_train, y_train)
        cv_acc_scores_trn.append(accuracy_trn)
        avg_score_trn = np.mean(cv_acc_scores_trn)
        
        accuracy_tst = test(model, X_test, y_test)
        cv_acc_scores_tst.append(accuracy_tst)
        avg_score_tst = np.mean(cv_acc_scores_tst)
        
    return avg_score_trn, avg_score_tst
