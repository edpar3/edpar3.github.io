# Modify the 'transform_feautures' function.
# This will solve the problem of null values.
# It will also create some new features.

def transform_features(df):
    
    # Drop columns that have no value for machine learning.
    df = df.drop(['Order', 'PID'], axis=1)
    
    # Drop columns with NA of more than 5 percent.
    null_series = df.isnull().sum()
    cols_dropping = null_series[(null_series > len(df) * 0.05)]
    df = df.drop(cols_dropping.index, axis=1)
    
    # Fill in remaining missing values with mode.
    # Note selection of first value for mode.
    # In this case all remaining columns have only one mode.
    for column in df.columns:
        df[column].fillna(df[column].mode()[0], inplace=True)
    
    # Create some new features.
    age_at_sale = df['Yr Sold'] - df['Year Built']
    years_since_remod = df['Yr Sold'] - df['Year Remod/Add']
    df['Age At Sale'] = age_at_sale
    df['Years Since Remod'] = years_since_remod
    
    # Drop rows based on unacceptable condition in 'Age at Sale'.
    index_names = df[ df['Age At Sale'] < 0 ].index
    df.drop(index_names, inplace = True)
    
    # Drop rows based on unacceptable condition in 'Years Since Remod'.
    index_names = df[ df['Years Since Remod'] < 0 ].index
    df.drop(index_names, inplace = True)
    
    # Drop other columns that leak info about the final sale.
    df = df.drop([
        'Mo Sold', 
        'Sale Condition', 
        'Sale Type',
        'Year Built',
        'Year Remod/Add', 
        'Yr Sold'], axis=1)
    
    return df

def select_features(df):
    
    # Select numerical features.
    num_housing = df.select_dtypes(include=['int', 'float'])

    sorted_corrs = corrmat['SalePrice'].abs().sort_values()

    return df

def train_and_test(df, k=0):
    
    if k == 0:
        
        # Shuffle the DataFrame.
        df = df.sample(frac=1, random_state=1).reset_index(drop=True)
        
        # Create train and test DataFrames.
        x = int((len(df) * 0.5))
        train = df.iloc[0:x]
        test = df.iloc[x:]
        
        # Instantiate a LinearRegression class.
        lr = LinearRegression()
        
        # Select features.
        features = train.columns.drop('SalePrice')
        
        # Fit the model.
        lr.fit(train[features], train.SalePrice)
        
        # Test the model and return RMSE.
        test_predictions = lr.predict(test.drop('SalePrice', axis=1))
        test_mse = mean_squared_error(test_predictions, test.SalePrice)
        test_rmse = np.sqrt(test_mse)
        
        return test_rmse
    
    if k == 1:
        
        # Shuffle the DataFrame.
        df = df.sample(frac=1, random_state=1).reset_index(drop=True)
        
        # Create folds for simple cross validation.
        if len(df)%2 == 0:
            x = int((len(df) * 0.5))
            fold_one = df.iloc[0:x]
            fold_two = df.iloc[x:]
        else:
            x = int((len(df) * 0.5))
            fold_one = df.iloc[0:(x)]
            fold_two = df.iloc[(x+1):]
        
        # Instantiate a LinearRegression class.
        lr = LinearRegression()
        
        # Train and test 'fold_one'.
        features_f1 = fold_one.columns.drop('SalePrice')
        lr.fit(fold_one[features_f1], fold_one.SalePrice)
        test_preds_f1 = lr.predict(fold_two.drop('SalePrice', axis=1))
        f1_mse = mean_squared_error(test_preds_f1, fold_two.SalePrice)
        f1_rmse = np.sqrt(f1_mse)
        
        # Train and test 'fold_two'.
        features_f2 = fold_two.columns.drop('SalePrice')
        lr.fit(fold_two[features_f2], fold_two.SalePrice)
        test_preds_f2 = lr.predict(fold_one.drop('SalePrice', axis=1))
        f2_mse = mean_squared_error(test_preds_f2, fold_one.SalePrice)
        f2_rmse = np.sqrt(f2_mse)
        
        # Calculate average rmse.
        avg_rmse = np.mean([f1_rmse, f2_rmse])
        
        return avg_rmse
    
    else:
        
        lr = LinearRegression()
        features = df.columns.drop('SalePrice')
        kf = KFold(n_splits=k, shuffle=True)
        rmse_values = []
        
        for train_index, test_index, in kf.split(df):
            train = df.iloc[train_index]
            test = df.iloc[test_index]
            lr.fit(train[features], train["SalePrice"])
            predictions = lr.predict(test[features])
            mse = mean_squared_error(test["SalePrice"], predictions)
            rmse = np.sqrt(mse)
            rmse_values.append(rmse)
        
        avg_rmse = np.mean(rmse_values)
        
        return avg_rmse
